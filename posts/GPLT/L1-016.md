## 思路

每个数字都有对应关系，可以用两个数组来存储这种关系，这样写比较简洁

## 代码
```c++
#include<bits/stdc++.h>
using namespace std;
int n;
bool pass;
int dx[]={7,9,10,5,8,4,2,1,6,3,7,9,10,5,8,4,2};
int dy[]={1,0,10,9,8,7,6,5,4,3,2};
void solve(){
	string s;
	cin>>s;
	int sum=0;
	for(int i=0;i<17;i++){
		if(s[i]>='0'&&s[i]<='9')sum=(sum+(s[i]-'0')*dx[i])%11;
		else {
			pass=true;
			cout<<s<<'\n';
			return ;
		}
	}
	char c;
	int t=dy[sum];
	if(t==10)c='X';
	else c=t+'0';
	if(c!=s[17]){
		pass=true;
		cout<<s<<'\n';
	}
}
int main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	cin>>n;
	while(n--){
		solve();
	}
	if(!pass)cout<<"All passed";
	return 0;
}
```
