很简单的题目，直接给代码
## 代码
```c++
#include<bits/stdc++.h>
using namespace std;
int a,b;
int main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	cin>>a>>b;
	int p=0;
	for(int i=a;i<=b;i++){
		if(p==5){
			printf("\n");
			p=0;
		}
		printf("%5d",i);
		p++;
	}
	printf("\nSum = %d",(b+a)*(b-a+1)/2);
	return 0;
}
```
