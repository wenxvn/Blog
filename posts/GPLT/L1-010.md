非常简单的比大小题

## 代码
```c++
#include<bits/stdc++.h>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	vector<int>s(3);
	for(int i=0;i<3;i++)cin>>s[i];
	sort(s.begin(),s.end());
	for(int i=0;i<3;i++){
		if(i)cout<<"->";
		cout<<s[i];
	}
	return 0;
}
```
