## 思路
需要不断枚举因子

可以先把第一个因子用for枚举出来，接下来的因子用另一个for枚举（因为因子是连续的），不断检验枚举出来的因子是否符合条件

注意开long long

## 代码
```c++
#include<bits/stdc++.h>
using namespace std;
#define int long long
int n;
signed main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	cin>>n;
	int ans=0;
	int start=n;
	for(int i=2;i*i<=n;i++){
		int sum=1;
		for(int j=i;;j++){
			sum*=j;
			if(sum>n||n%sum)break;
			int len=j-i+1;
			if(len>ans){
				ans=len;
				start=i;
			}
		}
	}
	if(ans){
		cout<<ans<<'\n';
		for(int i=0;i<ans;i++){
			if(i)cout<<"*";
			cout<<start+i;
		}
	}
	else cout<<1<<'\n'<<n;
	return 0;
}
```
