## 思路
1,7,17,31,49......

总共用掉了2*k*k-1个字符（k表示上半部分（含中间）的层数）

所以计算上半部分（含中间）一共有几层：层数k=sqrt((n+1)/2)

之后就只要根据层数打印空格和指定的字符即可

## 代码

```c++
#include<bits/stdc++.h>
using namespace std;
int n;
char c;
int main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	cin>>n>>c;
	int k=sqrt((n+1)/2);
	for(int i=0;i<k;i++){
		int cnt=(k-i)*2-1;
		for(int j=0;j<i;j++)cout<<" ";
		for(int j=0;j<cnt;j++)cout<<c;
		cout<<'\n';
	}
	for(int i=2;i<=k;i++){
		int cnt=i*2-1;
		for(int j=0;j<k-i;j++)cout<<" ";
		for(int j=0;j<cnt;j++)cout<<c;
		cout<<'\n';
	}
	cout <<n-(2*k*k-1)<<'\n';
	return 0;
}
```
