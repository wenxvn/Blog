很简单的题目，可以用string数组先存一下结果，这里直接给出代码
## 代码
```c++
#include<bits/stdc++.h>
using namespace std;
string s;
string a[]={"ling","yi","er","san","si","wu","liu","qi","ba","jiu"};
int main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	cin>>s;
	for(int i=0;i<s.size();i++){
		if(i)cout<<" ";
		if(s[i]=='-')cout<<"fu";
		else cout<<a[s[i]-'0'];
	}
	return 0;
}
```
