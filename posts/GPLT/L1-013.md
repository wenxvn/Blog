简单题

## 代码
```c++
#include<bits/stdc++.h>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	int n;cin>>n;
	int sum=1;
	int ans=1;
	for(int i=2;i<=n;i++){
		sum*=i;
		ans+=sum;
	}
	cout<<ans;
	return 0;
}
```
